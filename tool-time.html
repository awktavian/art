# Al's Proper Canvas Integration
import asyncio
import aiohttp
from ratelimit import limits, sleep_and_retry

# Respect rate limits (700 requests/minute)
@sleep_and_retry
@limits(calls=10, period=1)
async def canvas_request(session, method, endpoint, **kwargs):
    async with session.request(
        method, f"{BASE}{endpoint}", headers=headers, **kwargs
    ) as resp:
        if resp.status == 429:
            await asyncio.sleep(60)  # Back off gracefully
            return await canvas_request(session, method, endpoint, **kwargs)
        return await resp.json()

# Create quizzes properly â€” the Al-approved way
async def create_quiz_safely(course_id, quiz_data):
    async with aiohttp.ClientSession() as session:
        quiz = await canvas_request(
            session, "POST",
            f"/courses/{course_id}/quizzes",
            json={"quiz": quiz_data}
        )
        if quiz.get("id"):
            print(f"âœ“ Created: {quiz['title']}")
        return quiz
